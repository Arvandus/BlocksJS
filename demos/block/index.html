Block Demo

JavaScript

				    
(function () {
		
    var game, block;
				
    // Create a game
    game = BLOCKS.game();
    
    // Create an example block
    block = BLOCKS.block();
	
    block.addSlice({
        name: "default",
        layer: BLOCKS.layer(game.element), // Create a new canvas for the slice
        imageSrc: "images/coin.png", // Define sprite sheet location
        numberOfFrames: 10, // Define number of frames of animation
        loop: true
    });
    
    // Notify the block of game loop
    game.update = block.update;			
    game.render = block.render;
			
} ());
